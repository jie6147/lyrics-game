    let current = 0;

    const titleEl = document.getElementById("song-title");

    const playerEl = document.getElementById("youtube-player");

    const playBtn = document.getElementById("play-btn");

    const answerEl = document.getElementById("answer");

    const submitBtn = document.getElementById("submit-btn");

    const resultBox = document.getElementById("result-box");

    const correctEl = document.getElementById("correct-line");

    const prevEl = document.getElementById("prev-line");

    const nextEl = document.getElementById("next-line");

    const infoEl = document.getElementById("info");

    const nextBtn = document.getElementById("next-btn");

    const hintSection = document.getElementById("hint-section");

    const hintOutput = document.getElementById("hint-output");

    function loadQuestion() {

      const song = data[current];

      titleEl.textContent = song.title + " - " + song.artist;

      playerEl.innerHTML = "";

      playerEl.classList.add("hidden");

      resultBox.classList.add("hidden");

      hintOutput.textContent = "";

      hintSection.classList.add("hidden");

      answerEl.value = "";

    }

    playBtn.addEventListener("click", () => {

      const song = data[current];

      const embed = `<iframe src="https://www.youtube.com/embed/${song.youtubeId}?start=${song.start}&autoplay=1&controls=1" allow="autoplay"></iframe>`;

      playerEl.innerHTML = embed;

      playerEl.classList.remove("hidden");

    });

    submitBtn.addEventListener("click", () => {

      const song = data[current];

      const input = answerEl.value.trim();

      correctEl.textContent = song.correct;

      prevEl.textContent = song.prev;

      nextEl.textContent = song.next;

      infoEl.textContent = song.info;

      hintSection.classList.remove("hidden");

      resultBox.classList.remove("hidden");

    });

    document.querySelectorAll(".hint").forEach(btn => {

      btn.addEventListener("click", () => {

        const type = btn.dataset.type;

        const hint = data[current].hint[type];

        hintOutput.textContent = `提示：${hint}`;

      });

    });

    nextBtn.addEventListener("click", () => {

      current = (current + 1) % data.length;

      loadQuestion();

    });
